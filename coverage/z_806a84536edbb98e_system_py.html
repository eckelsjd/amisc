465 def sample_inputs(self, size: tuple | int, comp: str = 'System', use_pdf: bool = False, 

466 nominal: dict[str: float] = None, constants: set[str] = None) -> Dataset: 

467 """Return samples of the inputs according to provided options. Will return samples in the 

468 normalized/compressed space of the surrogate. See [`to_model_dataset`][amisc.utils.to_model_dataset] to 

469 convert the samples to be usable by the true model directly. 

470 

471 :param size: tuple or integer specifying shape or number of samples to obtain 

472 :param comp: which component to sample inputs for (defaults to full system exogenous inputs) 

473 :param use_pdf: whether to sample from each variable's pdf, defaults to random samples over input domain instead 

474 :param nominal: `dict(var_id=value)` of nominal values for params with relative uncertainty, also can use 

475 to specify constant values for a variable listed in `constants`. Specify nominal values 

476 as the true model would expect them (i.e. not normalized or compressed) -- they will be 

477 returned in normalized form. 

478 :param constants: set of param types to hold constant while sampling (i.e. calibration, design, etc.), 

479 can also put a `var.name` string in here to specify a single variable to hold constant 

480 :returns: `dict` of `(*size,)` samples for each input variable 

481 """ 

482 size = (size, ) if isinstance(size, int) else size 

483 nominal = nominal or dict() 

484 constants = constants or set() 

485 inputs = self.inputs() if comp == 'System' else self[comp].inputs 

486 samples = {} 

487 for var in inputs: 

488 # Sample from latent variable domains for field quantities 

489 if var.compression is not None: 

490 if var.category in constants or var in constants: 

491 nom = nominal.get(var, None) 

492 if nom is not None: 

493 nom = np.broadcast_to(np.atleast_1d(nom), size + (var.compression.latent_size(),)).copy() 

494 for i, d in enumerate(var.get_domain()): 

495 samples[f'{var.name}{LATENT_STR_ID}{i}'] = nom[..., i] if nom is not None else np.mean(d) 

496 else: 

497 latent = var.sample_domain(size) 

498 for i in range(latent.shape[-1]): 

499 samples[f'{var.name}{LATENT_STR_ID}{i}'] = latent[..., i] 

500 

501 # Sample scalars normally 

502 else: 

503 # Set a constant value for this variable 

504 if var.category in constants or var in constants: 

505 if nom := nominal.get(var): 

506 samples[var.name] = var.normalize(nom) 

507 else: 

508 samples[var.name] = var.normalize(var.get_nominal()) 

509 

510 # Sample from this variable's pdf or randomly within its domain bounds (reject if outside bounds) 

511 else: 

512 lb, ub = var.get_domain() 

513 x_sample = var.sample(size, nominal=nominal.get(var, None)) if use_pdf else var.sample_domain(size) 

514 good_idx = (x_sample < ub) & (x_sample > lb) 

515 num_reject = np.sum(~good_idx) 

516 

517 while num_reject > 0: 

518 new_sample = var.sample((num_reject,), nominal=nominal.get(var, None)) \ 

519 if use_pdf else var.sample_domain((num_reject,)) 

520 x_sample[~good_idx] = new_sample 

521 good_idx = (x_sample < ub) & (x_sample > lb) 

522 num_reject = np.sum(~good_idx) 

523 

524 samples[var.name] = var.normalize(x_sample) 

525 

526 return samples

528 def simulate_fit(self): 

529 """Loop back through training history and simulate each iteration. Will yield the internal data structures 

530 of each `Component` surrogate after each iteration of training (without needing to call `fit()` or any 

531 of the underlying models). This might be useful, for example, for computing the surrogate predictions on 

532 a new test set or viewing cumulative training costs. 

533 

534 !!! Example 

535 Say you have a new test set: `(new_xtest, new_ytest)`, and you want to compute the accuracy of the 

536 surrogate fit at each iteration of the training history: 

537 

538 ```python 

539 for train_iter, active_sets, candidate_sets, misc_coeff_train, misc_coeff_test in system.simulate_fit(): 

540 # Do something with the surrogate data structures 

541 new_ysurr = system.predict(new_xtest, index_set=active_sets, misc_coeff=misc_coeff_train) 

542 train_error = relative_error(new_ysurr, new_ytest) 

543 ``` 

544 

545 :return: a generator of the active index sets, candidate index sets, and MISC coefficients 

546 of each component model at each iteration of the training history 

547 """ 

548 # "Simulated" data structures for each component 

549 active_sets = {comp.name: IndexSet() for comp in self.components} # active index sets for each component 

550 candidate_sets = {comp.name: IndexSet() for comp in self.components} # candidate sets for each component 

551 misc_coeff_train = {comp.name: MiscTree() for comp in self.components} # MISC coeff for active sets 

552 misc_coeff_test = {comp.name: MiscTree() for comp in self.components} # MISC coeff for active + candidate sets 

553 

554 for train_result in self.train_history: 

555 # The selected refinement component and indices 

556 comp_star = train_result['component'] 

557 alpha_star = train_result['alpha'] 

558 beta_star = train_result['beta'] 

559 comp = self[comp_star] 

560 

561 # Get forward neighbors for the selected index 

562 neighbors = comp._neighbors(alpha_star, beta_star, active_set=active_sets[comp_star], forward=True) 

563 

564 # "Activate" the index in the simulated data structure 

565 s = set() 

566 s.add((alpha_star, beta_star)) 

567 comp.update_misc_coeff(IndexSet(s), index_set=active_sets[comp_star],

575 misc_coeff=misc_coeff_test[comp_star])

578 comp.update_misc_coeff(neighbors, index_set=active_sets[comp_star].union(candidate_sets[comp_star]), 

579 misc_coeff=misc_coeff_test[comp_star]) # neighbors will only ever pass here once 

580 candidate_sets[comp_star].update(neighbors)

583 # See the "index_set" and "misc_coeff" overrides for `System.predict()` for example

610 """Train the system surrogate adaptively by iterative refinement until an end condition is met. 

611 

612 :param targets: list of system output variables to focus refinement on, use all outputs if not specified 

613 :param num_refine: number of input samples to compute error indicators on 

614 :param max_iter: the maximum number of refinement steps to take 

615 :param max_tol: the max allowable value in relative L2 error to achieve 

616 :param runtime_hr: the threshold wall clock time (hr) at which to stop further refinement (will go 

617 until all models finish the current iteration) 

618 :param save_interval: number of refinement steps between each progress save, none if 0; `System.root_dir` 

619 must be specified to save to file 

620 :param estimate_bounds: whether to estimate bounds for the coupling variables; will only try to estimate from 

621 the `test_set` if provided (defaults to `True`). Otherwise, you should manually 

622 provide domains for all coupling variables. 

623 :param update_bounds: whether to continuously update coupling variable bounds during refinement 

624 :param test_set: `tuple` of `(xtest, ytest)` to show convergence of surrogate to the true model. The test set 

625 inputs and outputs are specified as `dicts` of `np.ndarrays` with keys corresponding to the 

626 variable names. Can also pass a path to a `.pkl` file that has the test set data as 

627 {'test_set': (xtest, ytest)}. 

628 :param start_test_check: the iteration to start checking the test set error (defaults to twice the number

630 should at least allow one iteration per component before checking test set error 

631 :param plot_interval: how often to plot the error indicator and test set error (defaults to every iteration); 

632 will only plot and save to file if a root directory is set

634 recommended for expensive models)

637 targets = targets or self.outputs()

644 if estimate_bounds:

653 if var.compression is not None: 

654 new_domain = list(zip(y_min[var].tolist(), y_max[var].tolist())) 

655 var.update_domain(new_domain, override=True) 

656 else: 

657 new_domain = (float(y_min[var]), float(y_max[var])) 

658 var.update_domain(var.denormalize(new_domain), override=True) 

659 del y_samples 

660 else: 

661 self.logger.warning('Could not estimate bounds for coupling variables: no test set provided. ' 

662 'Make sure you manually provide (good) coupling variable domains.') 

663 

664 # Track convergence progress on the error indicator and test set (plot to file) 

665 if self.root_dir is not None: 

666 err_record = [res['added_error'] for res in self.train_history] 

667 err_fig, err_ax = plt.subplots(figsize=(6, 5), layout='tight') 

668 

669 if xtest is not None and ytest is not None: 

670 num_plot = min(len(targets), 3) 

671 test_record = np.full((self.refine_level, num_plot), np.nan) 

672 t_fig, t_ax = plt.subplots(1, num_plot, figsize=(3.5 * num_plot, 4), layout='tight', squeeze=False, 

673 sharey='row') 

674 for j, res in enumerate(self.train_history): 

675 for i, var in enumerate(targets[:num_plot]): 

676 if (perf := res.get('test_error')) is not None: 

677 test_record[j, i] = perf[var] 

678 

679 while True: 

680 # Adaptive refinement step 

681 train_result = self.refine(targets=targets, num_refine=num_refine, update_bounds=update_bounds, 

682 executor=executor) 

683 if train_result['component'] is None: 

684 self._print_title_str('Termination criteria reached: No candidates left to refine') 

685 break 

686 

687 curr_error = train_result['added_error'] 

688 

689 # Plot progress of error indicator 

690 if self.root_dir is not None: 

691 err_record.append(curr_error) 

692 

693 if plot_interval > 0 and self.refine_level % plot_interval == 0: 

694 err_ax.clear(); err_ax.set_yscale('log'); err_ax.grid() 

695 err_ax.plot(err_record, '-k') 

696 err_ax.set_xlabel('Iteration'); err_ax.set_ylabel('Relative error indicator') 

697 err_fig.savefig(str(Path(self.root_dir) / 'error_indicator.pdf'), format='pdf', bbox_inches='tight') 

698 

699 # Save performance on a test set 

700 if xtest is not None and ytest is not None: 

701 perf = self.test_set_performance(xtest, ytest) if self.refine_level >= start_test_check else ( 

702 {str(var): np.nan for var in ytest}) # don't compute if components are uninitialized 

703 train_result['test_error'] = perf.copy() 

704 

705 if self.root_dir is not None: 

706 test_record = np.vstack((test_record, np.array([perf[var] for var in targets[:num_plot]]))) 

707 

708 if plot_interval > 0 and self.refine_level % plot_interval == 0: 

709 for i in range(num_plot): 

710 t_ax[0, i].clear(); t_ax[0, i].set_yscale('log'); t_ax[0, i].grid() 

711 t_ax[0, i].plot(test_record[:, i], '-k') 

712 t_ax[0, i].set_title(self.outputs()[targets[i]].get_tex(units=True)) 

713 t_ax[0, i].set_xlabel('Iteration') 

714 t_ax[0, i].set_ylabel('Test set relative error' if i==0 else '') 

715 t_fig.savefig(str(Path(self.root_dir) / 'test_set_error.pdf'),format='pdf',bbox_inches='tight') 

716 

717 self.train_history.append(train_result) 

718 if self.root_dir is not None and save_interval > 0 and self.refine_level % save_interval == 0: 

719 iter_name = f'{self.name}_iter{self.refine_level}' 

720 if not (pth := self.root_dir / 'surrogates' / iter_name).is_dir(): 

721 os.mkdir(pth) 

722 self.save_to_file(f'{iter_name}.yml', save_dir=pth) # Save to an iteration-specific directory 

723 

724 # Check all end conditions 

725 if self.refine_level >= max_iter: 

726 self._print_title_str(f'Termination criteria reached: Max iteration {self.refine_level}/{max_iter}') 

727 break 

728 if curr_error < max_tol: 

729 self._print_title_str(f'Termination criteria reached: relative error {curr_error} < tol {max_tol}') 

730 break 

731 if ((time.time() - t_start) / 3600.0) >= runtime_hr: 

732 actual = datetime.timedelta(seconds=time.time() - t_start) 

733 target = datetime.timedelta(seconds=runtime_hr * 3600) 

734 self._print_title_str(f'Termination criteria reached: runtime {str(actual)} > {str(target)}') 

735 break 

736 

737 if self.root_dir is not None: 

738 iter_name = f'{self.name}_iter{self.refine_level}' 

739 if not (pth := self.root_dir / 'surrogates' / iter_name).is_dir(): 

740 os.mkdir(pth) 

741 self.save_to_file(f'{iter_name}.yml', save_dir=pth) 

742 

743 if xtest is not None and ytest is not None: 

744 self._save_test_set((xtest, ytest)) 

745 

746 self.logger.info(f'Final system surrogate: \n {self}')

761 def refine(self, targets: list = None, num_refine: int = 100, update_bounds: bool = True, 

762 executor: Executor = None) -> TrainIteration: 

763 """Perform a single adaptive refinement step on the system surrogate. 

764 

765 :param targets: list of system output variables to focus refinement on, use all outputs if not specified 

766 :param num_refine: number of input samples to compute error indicators on 

767 :param update_bounds: whether to continuously update coupling variable bounds during refinement 

768 :param executor: a `concurrent.futures.Executor` object to parallelize model evaluations 

769 :returns: `dict` of the refinement results indicating the chosen component and candidate index 

770 """ 

771 self._print_title_str(f'Refining system surrogate: iteration {self.refine_level + 1}') 

772 targets = targets or self.outputs() 

773 

774 # Check for uninitialized components and refine those first 

775 for comp in self.components: 

776 if len(comp.active_set) == 0 and comp.has_surrogate: 

777 alpha_star = (0,) * len(comp.model_fidelity) 

778 beta_star = (0,) * len(comp.max_beta) 

779 self.logger.info(f"Initializing component {comp.name}: adding {(alpha_star, beta_star)} to active set") 

780 model_dir = (pth / 'components' / comp.name) if (pth := self.root_dir) is not None else None 

781 comp.activate_index(alpha_star, beta_star, model_dir=model_dir, executor=executor) 

782 cost_star = max(1., comp.get_cost(alpha_star, beta_star)) # Cpu time (s) 

783 err_star = np.nan 

784 num_evals = round(cost_star / comp.model_costs.get(alpha_star, 1.)) 

785 return {'component': comp.name, 'alpha': alpha_star, 'beta': beta_star, 'num_evals': int(num_evals), 

786 'added_cost': float(cost_star), 'added_error': float(err_star)} 

787 

788 # Compute entire integrated-surrogate on a random test set for global system QoI error estimation 

789 x_samples = self.sample_inputs(num_refine) 

790 y_curr = self.predict(x_samples, index_set='train', targets=targets) 

791 _combine_latent_arrays(y_curr) 

792 coupling_vars = {k: v for k, v in self.coupling_variables().items() if k in y_curr} 

793 

794 y_min, y_max = None, None 

795 if update_bounds: 

796 y_min = {var: np.nanmin(y_curr[var], axis=0, keepdims=True) for var in coupling_vars} # (1, ydim) 

797 y_max = {var: np.nanmax(y_curr[var], axis=0, keepdims=True) for var in coupling_vars} # (1, ydim) 

798 

799 # Find the candidate surrogate with the largest error indicator 

800 error_max, error_indicator = -np.inf, -np.inf 

801 comp_star, alpha_star, beta_star, err_star, cost_star = None, None, None, -np.inf, 0 

802 for comp in self.components: 

803 if not comp.has_surrogate: # Skip analytic models that don't need a surrogate 

804 continue 

805 

806 self.logger.info(f"Estimating error for component '{comp.name}'...") 

807 

808 if len(comp.candidate_set) > 0: 

809 candidates = list(comp.candidate_set) 

810 if executor is None: 

811 ret = [self.predict(x_samples, targets=targets, index_set={comp.name: {(alpha, beta)}}, 

812 incremental={comp.name: True}) 

813 for alpha, beta in candidates] 

814 else: 

815 temp_buffer = self._remove_unpickleable() 

816 futures = [executor.submit(self.predict, x_samples, targets=targets, 

817 index_set={comp.name: {(alpha, beta)}}, incremental={comp.name: True}) 

818 for alpha, beta in candidates] 

819 wait(futures, timeout=None, return_when=ALL_COMPLETED) 

820 ret = [f.result() for f in futures] 

821 self._restore_unpickleable(temp_buffer) 

822 

823 for i, y_cand in enumerate(ret): 

824 alpha, beta = candidates[i] 

825 _combine_latent_arrays(y_cand) 

826 error = {} 

827 for var, arr in y_cand.items(): 

828 if var in targets: 

829 error[var] = relative_error(arr, y_curr[var]) 

830 

831 if update_bounds and var in coupling_vars: 

832 y_min[var] = np.nanmin(np.concatenate((y_min[var], arr), axis=0), axis=0, keepdims=True) 

833 y_max[var] = np.nanmax(np.concatenate((y_max[var], arr), axis=0), axis=0, keepdims=True) 

834 

835 delta_error = np.nanmax([np.nanmax(error[var]) for var in error]) # Max error over all target QoIs 

836 delta_work = max(1., comp.get_cost(alpha, beta)) # Cpu time (s) 

837 error_indicator = delta_error / delta_work 

838 

839 self.logger.info(f"Candidate multi-index: {(alpha, beta)}. Relative error: {delta_error}. " 

840 f"Error indicator: {error_indicator}.") 

841 

842 if error_indicator > error_max: 

843 error_max = error_indicator 

844 comp_star, alpha_star, beta_star, err_star, cost_star = ( 

845 comp.name, alpha, beta, delta_error, delta_work) 

846 else: 

847 self.logger.info(f"Component '{comp.name}' has no available candidates left!") 

848 

849 # Update all coupling variable ranges 

850 if update_bounds: 

851 for var in coupling_vars.values(): 

852 if np.all(~np.isnan(y_min[var])) and np.all(~np.isnan(y_max[var])): 

853 if var.compression is not None: 

854 new_domain = list(zip(np.squeeze(y_min[var], axis=0).tolist(), 

855 np.squeeze(y_max[var], axis=0).tolist())) 

856 var.update_domain(new_domain) 

857 else: 

858 new_domain = (y_min[var][0], y_max[var][0]) 

859 var.update_domain(var.denormalize(new_domain)) # bds will be in norm space from predict() call 

860 

861 # Add the chosen multi-index to the chosen component 

862 if comp_star is not None: 

863 self.logger.info(f"Candidate multi-index {(alpha_star, beta_star)} chosen for component '{comp_star}'.") 

864 model_dir = (pth / 'components' / comp_star) if (pth := self.root_dir) is not None else None 

865 self[comp_star].activate_index(alpha_star, beta_star, model_dir=model_dir, executor=executor) 

866 num_evals = round(cost_star / self[comp_star].model_costs.get(alpha_star, 1.)) 

867 else: 

868 self.logger.info(f"No candidates left for refinement, iteration: {self.refine_level}") 

869 num_evals = 0 

870 

871 # Return the results of the refinement step 

872 return {'component': comp_star, 'alpha': alpha_star, 'beta': beta_star, 'num_evals': int(num_evals), 

873 'added_cost': float(cost_star), 'added_error': float(err_star)}

875 def predict(self, x: dict | Dataset, 

876 max_fpi_iter: int = 100, 

877 anderson_mem: int = 10, 

878 fpi_tol: float = 1e-10, 

879 use_model: str | tuple | dict = None, 

880 model_dir: str | Path = None, 

881 verbose: bool = False, 

882 index_set: dict[str: IndexSet | Literal['train', 'test']] = 'test', 

883 misc_coeff: dict[str: MiscTree] = None, 

884 normalized_inputs: bool = True, 

885 incremental: dict[str, bool] = False, 

886 targets: list[str] = None, 

887 executor: Executor = None, 

888 var_shape: dict[str, tuple] = None) -> Dataset: 

889 """Evaluate the system surrogate at inputs `x`. Return `y = system(x)`. 

890 

891 !!! Warning "Computing the true model with feedback loops" 

892 You can use this function to predict outputs for your MD system using the full-order models rather than the 

893 surrogate, by specifying `use_model`. This is convenient because the `System` manages all the 

894 coupled information flow between models automatically. However, it is *highly* recommended to not use 

895 the full model if your system contains feedback loops. The FPI nonlinear solver would be infeasible using 

896 anything more computationally demanding than the surrogate. 

897 

898 :param x: `dict` of input samples for each variable in the system 

899 :param max_fpi_iter: the limit on convergence for the fixed-point iteration routine 

900 :param anderson_mem: hyperparameter for tuning the convergence of FPI with anderson acceleration 

901 :param fpi_tol: tolerance limit for convergence of fixed-point iteration 

902 :param use_model: 'best'=highest-fidelity, 'worst'=lowest-fidelity, tuple=specific fidelity, None=surrogate, 

903 specify a `dict` of the above to assign different model fidelities for diff components 

904 :param model_dir: directory to save model outputs if `use_model` is specified 

905 :param verbose: whether to print out iteration progress during execution 

906 :param index_set: `dict(comp=[indices])` to override the active set for a component, defaults to using the 

907 `test` set for every component. Can also specify `train` for any component or a valid 

908 `IndexSet` object. If `incremental` is specified, will be overwritten with `train`. 

909 :param misc_coeff: `dict(comp=MiscTree)` to override the default coefficients for a component, passes through 

910 along with `index_set` and `incremental` to `comp.predict()`. 

911 :param normalized_inputs: true if the passed inputs are compressed/normalized for surrogate evaluation 

912 (default), such as inputs returned by `sample_inputs`. Set to `False` if you are 

913 passing inputs as the true models would expect them instead (i.e. not normalized). 

914 :param incremental: whether to add `index_set` to the current active set for each component (temporarily); 

915 this will set `index_set='train'` for all other components (since incremental will 

916 augment the "training" active sets, not the "testing" candidate sets) 

917 :param targets: list of output variables to return, defaults to returning all system outputs 

918 :param executor: a `concurrent.futures.Executor` object to parallelize model evaluations 

919 :param var_shape: (Optional) `dict` of shapes for field quantity inputs in `x` -- you would only specify this 

920 if passing field qtys directly to the models (i.e. not using `sample_inputs`) 

921 :returns: `dict` of output variables - the surrogate approximation of the system outputs (or the true model) 

922 """ 

923 # Format inputs and allocate space 

924 var_shape = var_shape or {} 

925 x, loop_shape = format_inputs(x, var_shape=var_shape) # {'x': (N, *var_shape)} 

926 y = {} 

927 all_inputs = ChainMap(x, y) 

928 N = int(np.prod(loop_shape)) 

929 t1 = 0 

930 output_dir = None 

931 norm_status = {var: normalized_inputs for var in x} # keep track of whether inputs are normalized or not 

932 graph = self.graph() 

933 

934 # Keep track of what outputs are computed 

935 is_computed = {} 

936 for var in (targets or self.outputs()): 

937 if (v := self.outputs().get(var, None)) is not None: 

938 if v.compression is not None: 

939 for field in v.compression.fields: 

940 is_computed[field] = False 

941 else: 

942 is_computed[var] = False

944 def _set_default(struct: dict, default=None): 

945 """Helper to set a default value for each component key in a `dict`. Ensures all components have a value.""" 

946 if struct is not None: 

947 if not isinstance(struct, dict): 

948 struct = {node: struct for node in graph.nodes} # use same for each component 

949 else: 

950 struct = {node: default for node in graph.nodes} 

951 return {node: struct.get(node, default) for node in graph.nodes} 

952 

953 # Ensure use_model, index_set, and incremental are specified for each component model 

954 use_model = _set_default(use_model, None) 

955 incremental = _set_default(incremental, False) # default to train if incremental anywhere 

956 index_set = _set_default(index_set, 'train' if any([incremental[node] for node in graph.nodes]) else 'test') 

957 misc_coeff = _set_default(misc_coeff, None)

961 def _gather_comp_inputs(comp, coupling=None): 

962 """Helper to gather inputs for a component, making sure they are normalized correctly. Any coupling 

963 variables passed in will be used in preference over `all_inputs`. 

964 """ 

965 # Will access but not modify: all_inputs, use_model, norm_status 

966 kwds = {} 

967 comp_input = {} 

968 coupling = coupling or {} 

969 

970 # Take coupling variables as a priority 

971 comp_input.update({var: np.copy(arr[samples.curr_idx, ...]) for var, arr in 

972 coupling.items() if str(var).split(LATENT_STR_ID)[0] in comp.inputs}) 

973 # Gather all other inputs 

974 for var, arr in all_inputs.items(): 

975 var_id = str(var).split(LATENT_STR_ID)[0] 

976 if var_id in comp.inputs and var not in coupling: 

977 comp_input[var] = np.copy(arr[samples.curr_idx, ...]) 

978 

979 # Gather extra fields (will never be in coupling since field couplings should always be latent coeff) 

980 field_coords = {f'{var}_coords': comp.model_kwargs.get(f'{var}_coords', None) for var in comp.inputs} 

981 for var in comp.inputs: 

982 if var not in comp_input and var.compression is not None: 

983 for field in var.compression.fields: 

984 if field in all_inputs: 

985 comp_input[field] = np.copy(all_inputs[field][samples.curr_idx, ...]) 

986 

987 call_model = use_model.get(comp.name, None) is not None 

988 

989 # Make sure we format all inputs for model evaluation (i.e. denormalize) 

990 if call_model: 

991 norm_inputs = {var: arr for var, arr in comp_input.items() if norm_status[var]} 

992 if len(norm_inputs) > 0: 

993 denorm_inputs, field_coords = to_model_dataset(norm_inputs, comp.inputs, del_latent=True, 

994 **field_coords) 

995 for var in norm_inputs: 

996 del comp_input[var] 

997 kwds.update(field_coords) 

998 comp_input.update(denorm_inputs) 

999 

1000 # Otherwise, make sure we format inputs for surrogate evaluation (i.e. normalize) 

1001 else: 

1002 denorm_inputs = {var: arr for var, arr in comp_input.items() if not norm_status[var]} 

1003 if len(denorm_inputs) > 0: 

1004 norm_inputs, _ = to_surrogate_dataset(denorm_inputs, comp.inputs, del_fields=True, **field_coords) 

1005 

1006 for var in denorm_inputs: 

1007 del comp_input[var] 

1008 comp_input.update(norm_inputs) 

1009 

1010 return comp_input, kwds, call_model 

1011 

1012 # Convert system into DAG by grouping strongly-connected-components 

1013 dag = nx.condensation(graph) 

1014 

1015 # Compute component models in topological order 

1016 for supernode in nx.topological_sort(dag): 

1017 if np.all(list(is_computed.values())): 

1018 break # Exit early if all selected return qois are computed 

1019 

1020 scc = [n for n in dag.nodes[supernode]['members']] 

1021 samples.reset_convergence() 

1022 

1023 # Compute single component feedforward output (no FPI needed) 

1024 if len(scc) == 1: 

1025 if verbose: 

1026 self.logger.info(f"Running component '{scc[0]}'...") 

1027 t1 = time.time() 

1028 

1029 # Gather inputs 

1030 comp = self[scc[0]] 

1031 comp_input, kwds, call_model = _gather_comp_inputs(comp) 

1032 

1033 # Compute outputs 

1034 if model_dir is not None: 

1035 output_dir = Path(model_dir) / scc[0] 

1036 if not output_dir.exists(): 

1037 os.mkdir(output_dir) 

1038 comp_output = comp.predict(comp_input, use_model=use_model.get(scc[0]), model_dir=output_dir, 

1039 index_set=index_set.get(scc[0]), incremental=incremental.get(scc[0]), 

1040 misc_coeff=misc_coeff.get(scc[0]), executor=executor, **kwds) 

1041 for var, arr in comp_output.items(): 

1042 output_shape = arr.shape[1:] 

1043 if y.get(var) is None: 

1044 y.setdefault(var, np.full((N, *output_shape), np.nan)) 

1045 y[var][samples.curr_idx, ...] = arr 

1046 samples.valid_idx = np.logical_and(samples.valid_idx, ~np.any(np.isnan(y[var]), 

1047 axis=tuple(range(1, y[var].ndim)))) 

1048 is_computed[str(var).split(LATENT_STR_ID)[0]] = True 

1049 norm_status[var] = not call_model 

1050 

1051 if verbose: 

1052 self.logger.info(f"Component '{scc[0]}' completed. Runtime: {time.time() - t1} s") 

1053 

1054 # Handle FPI for SCCs with more than one component 

1055 else: 

1056 # Set the initial guess for all coupling vars (middle of domain) 

1057 scc_inputs = ChainMap(*[self[comp].inputs for comp in scc]) 

1058 scc_outputs = ChainMap(*[self[comp].outputs for comp in scc]) 

1059 coupling_vars = [scc_inputs.get(var) for var in (scc_inputs.keys() - x.keys()) if var in scc_outputs] 

1060 coupling_prev = {} 

1061 for var in coupling_vars: 

1062 domain = var.get_domain() 

1063 if isinstance(domain, list): # Latent coefficients are the coupling variables 

1064 for i, d in enumerate(domain): 

1065 lb, ub = d 

1066 coupling_prev[f'{var.name}{LATENT_STR_ID}{i}'] = np.broadcast_to((lb + ub) / 2, (N,)).copy() 

1067 norm_status[f'{var.name}{LATENT_STR_ID}{i}'] = True 

1068 else: 

1069 lb, ub = var.normalize(domain) 

1070 shape = (N,) + (1,) * len(var_shape.get(var, ())) 

1071 coupling_prev[var] = np.broadcast_to((lb + ub) / 2, shape).copy() 

1072 norm_status[var] = True 

1073 

1074 residual_hist = deque(maxlen=anderson_mem) 

1075 coupling_hist = deque(maxlen=anderson_mem) 

1076 

1077 def _end_conditions_met(): 

1078 """Helper to compute residual, update history, and check end conditions.""" 

1079 residual = {} 

1080 converged_idx = np.full(N, True) 

1081 for var in coupling_prev: 

1082 residual[var] = y[var] - coupling_prev[var] 

1083 var_conv = np.all(np.abs(residual[var]) <= fpi_tol, axis=tuple(range(1, residual[var].ndim))) 

1084 converged_idx = np.logical_and(converged_idx, var_conv) 

1085 samples.valid_idx = np.logical_and(samples.valid_idx, ~np.isnan(coupling_prev[var])) 

1086 samples.converged_idx = np.logical_or(samples.converged_idx, converged_idx) 

1087 

1088 for var in coupling_prev: 

1089 coupling_prev[var][samples.curr_idx, ...] = y[var][samples.curr_idx, ...] 

1090 residual_hist.append(copy.deepcopy(residual)) 

1091 coupling_hist.append(copy.deepcopy(coupling_prev)) 

1092 

1093 if int(np.sum(samples.curr_idx)) == 0: 

1094 if verbose: 

1095 self.logger.info(f'FPI converged for SCC {scc} in {k} iterations with tol ' 

1096 f'{fpi_tol}. Final time: {time.time() - t1} s') 

1097 return True 

1098 

1099 max_error = np.max([np.max(np.abs(res[samples.curr_idx, ...])) for res in residual.values()]) 

1100 if verbose: 

1101 self.logger.info(f'FPI iter: {k}. Max residual: {max_error}. Time: {time.time() - t1} s') 

1102 

1103 if k >= max_fpi_iter: 

1104 self.logger.warning(f'FPI did not converge in {max_fpi_iter} iterations for SCC {scc}: ' 

1105 f'{max_error} > tol {fpi_tol}. Some samples will be returned as NaN.') 

1106 for var in coupling_prev: 

1107 y[var][~samples.converged_idx, ...] = np.nan 

1108 samples.valid_idx = np.logical_and(samples.valid_idx, samples.converged_idx) 

1109 return True 

1110 else: 

1111 return False 

1112 

1113 # Main FPI loop 

1114 if verbose: 

1115 self.logger.info(f"Initializing FPI for SCC {scc} ...") 

1116 t1 = time.time() 

1117 k = 0 

1118 while True: 

1119 for node in scc: 

1120 # Gather inputs from exogenous and coupling sources 

1121 comp = self[node] 

1122 comp_input, kwds, call_model = _gather_comp_inputs(comp, coupling=coupling_prev) 

1123 

1124 # Compute outputs (just don't do this FPI with expensive real models, please..) 

1125 comp_output = comp.predict(comp_input, use_model=use_model.get(node), model_dir=None, 

1126 index_set=index_set.get(node), incremental=incremental.get(node), 

1127 misc_coeff=misc_coeff.get(node), executor=executor, **kwds) 

1128 for var, arr in comp_output.items(): 

1129 output_shape = arr.shape[1:] 

1130 if y.get(var) is not None: 

1131 if output_shape != y.get(var).shape[1:]: 

1132 y[var] = _merge_shapes((N, *output_shape), y[var]) 

1133 else: 

1134 y.setdefault(var, np.full((N, *output_shape), np.nan)) 

1135 norm_status[var] = not call_model 

1136 is_computed[var] = True 

1137 y[var][samples.curr_idx, ...] = arr 

1138 

1139 # Compute residual and check end conditions 

1140 if _end_conditions_met(): 

1141 break 

1142 

1143 # Skip anderson acceleration on first iteration 

1144 if k == 0: 

1145 k += 1 

1146 continue 

1147 

1148 # Iterate with anderson acceleration (only iterate on samples that are not yet converged) 

1149 N_curr = int(np.sum(samples.curr_idx)) 

1150 mk = len(residual_hist) # Max of anderson mem 

1151 var_shapes = [] 

1152 xdims = [] 

1153 for var in coupling_prev: 

1154 shape = coupling_prev[var].shape[1:] 

1155 var_shapes.append(shape) 

1156 xdims.append(int(np.prod(shape))) 

1157 N_couple = int(np.sum(xdims)) 

1158 res_snap = np.empty((N_curr, N_couple, mk)) # Shortened snapshot of residual history 

1159 coupling_snap = np.empty((N_curr, N_couple, mk)) # Shortened snapshot of coupling history 

1160 for i, (coupling_iter, residual_iter) in enumerate(zip(coupling_hist, residual_hist)): 

1161 start_idx = 0 

1162 for j, var in enumerate(coupling_prev): 

1163 end_idx = start_idx + xdims[j] 

1164 coupling_snap[:, start_idx:end_idx, i] = coupling_iter[var][samples.curr_idx, ...].reshape((N_curr, -1)) # noqa: E501 

1165 res_snap[:, start_idx:end_idx, i] = residual_iter[var][samples.curr_idx, ...].reshape((N_curr, -1)) # noqa: E501 

1166 start_idx = end_idx 

1167 C = np.ones((N_curr, 1, mk)) 

1168 b = np.zeros((N_curr, N_couple, 1)) 

1169 d = np.ones((N_curr, 1, 1)) 

1170 alpha = np.expand_dims(constrained_lls(res_snap, b, C, d), axis=-3) # (..., 1, mk, 1) 

1171 coupling_new = np.squeeze(coupling_snap[:, :, np.newaxis, :] @ alpha, axis=(-1, -2)) 

1172 start_idx = 0 

1173 for j, var in enumerate(coupling_prev): 

1174 end_idx = start_idx + xdims[j] 

1175 coupling_prev[var][samples.curr_idx, ...] = coupling_new[:, start_idx:end_idx].reshape((N_curr, *var_shapes[j])) # noqa: E501 

1176 start_idx = end_idx 

1177 k += 1 

1178 

1179 # Return all component outputs; samples that didn't converge during FPI are left as np.nan 

1180 return format_outputs(y, loop_shape)

1314 def plot_slice(self, inputs: list[str] = None,

1323 from_file: str | Path = None, 

1324 subplot_size_in: float = 3.): 

1325 """Helper function to plot 1d slices of the surrogate and/or model outputs over the inputs. A single

1327 inputs constant at their nominal values (or smoothly varying them if `random_walk=True`). 

1328 This function is useful for visualizing the behavior of the system surrogate and/or model(s) over a 

1329 single input variable at a time.

1334 of `num_steps*len(inputs)` model/surrogate evaluations

1337 :param model_dir: base directory to save model outputs (if specified) 

1338 :param executor: a `concurrent.futures.Executor` object to parallelize model or surrogate evaluations 

1339 :param nominal: `dict` of `var->nominal` to use as constant values for all non-sliced variables (use

1348 input_slices, output_slices_model, output_slices_surr = None, None, None

1414 

1415 input_slices[var] = arr[..., np.newaxis] if input_slices.get(var) is None else ( 

1416 np.concatenate((input_slices[var], arr[..., np.newaxis]), axis=-1))

1458 

1459 ylabel = ylabels[i] if j == 0 else '' 

1460 xlabel = xlabels[j] if i == len(outputs) - 1 else '' 

1461 legend = (i == 0 and j == len(inputs) - 1)

1463 fig.set_size_inches(subplot_size_in * len(inputs), subplot_size_in * len(outputs)) 

1464 fig.tight_layout() 

1465 

1466 # Save results (unless we were already loading from a save file)

1480 def get_allocation(self, idx: int = None): 

1481 """Get a breakdown of cost allocation up to a certain iteration number during training (starting at 1). 

1482 

1483 :param idx: the iteration number to get allocation results for (defaults to last refinement step) 

1484 :returns: `cost_alloc, eval_alloc, cost_cum, eval_cum` - the cost allocation per model/fidelity, the 

1485 number of model evaluations per model/fidelity, the cumulative training cost, and the cumulative 

1486 number of model evaluations 

1487 """ 

1488 if idx is None: 

1489 idx = self.refine_level 

1490 if idx > self.refine_level: 

1491 raise ValueError(f'Specified index: {idx} is greater than the max training level of {self.refine_level}') 

1492 

1493 eval_alloc = dict() # Model evaluation counts per node and model fidelity 

1494 cost_alloc = dict() # Cost allocation (cpu time in s) per node and model fidelity 

1495 cost_cum = [] # Cumulative cost allocation during training 

1496 eval_cum = [] # Cumulative number of model evaluations during training 

1497 

1498 prev_cands = {comp.name: IndexSet() for comp in self.components} # empty candidate sets 

1499 

1500 # Add cumulative training costs 

1501 for train_res, active_sets, cand_sets, misc_coeff_train, misc_coeff_test in self.simulate_fit(): 

1502 comp = train_res['component'] 

1503 alpha = train_res['alpha'] 

1504 beta = train_res['beta'] 

1505 

1506 eval_alloc.setdefault(comp, dict()) 

1507 cost_alloc.setdefault(comp, dict()) 

1508 

1509 new_cands = cand_sets[comp].union({(alpha, beta)}) - prev_cands[comp] # newly computed candidates 

1510 

1511 iter_cost = 0. 

1512 iter_eval = 0 

1513 for alpha_new, beta_new in new_cands: 

1514 eval_alloc[comp].setdefault(alpha_new, 0.) 

1515 cost_alloc[comp].setdefault(alpha_new, 0.) 

1516 

1517 added_cost = self[comp].get_cost(alpha_new, beta_new) 

1518 model_cost = self[comp].model_costs.get(alpha_new, 1.) 

1519 added_eval = round(added_cost / model_cost) 

1520 

1521 iter_cost += added_cost 

1522 iter_eval += added_eval 

1523 

1524 eval_alloc[comp][alpha_new] += added_eval 

1525 cost_alloc[comp][alpha_new] += added_cost 

1526 

1527 cost_cum.append(iter_cost) 

1528 eval_cum.append(iter_eval) 

1529 prev_cands[comp] = cand_sets[comp].union({(alpha, beta)}) 

1530 

1531 return cost_alloc, eval_alloc, np.cumsum(cost_cum), np.cumsum(eval_cum)

1533 def plot_allocation(self, cmap: str = 'Blues', text_bar_width: float = 0.06, arrow_bar_width: float = 0.02): 

1534 """Plot bar charts showing cost allocation during training. 

1535 

1536 !!! Warning "Beta feature" 

1537 This has pretty good default settings, but it might look terrible for your use. Mostly provided here as 

1538 a template for making cost allocation bar charts. Please feel free to copy and edit in your own code. 

1539 

1540 :param cmap: the colormap string identifier for `plt` 

1541 :param text_bar_width: the minimum total cost fraction above which a bar will print centered model fidelity text 

1542 :param arrow_bar_width: the minimum total cost fraction above which a bar will try to print text with an arrow; 

1543 below this amount, the bar is too skinny and won't print any text 

1544 :returns: `fig, ax`, Figure and Axes objects 

1545 """ 

1546 # Get total cost (including offline overhead) 

1547 cost_alloc, eval_alloc, cost_cum, _ = self.get_allocation() 

1548 total_cost = cost_cum[-1] 

1549 

1550 # Remove nodes with cost=0 from alloc dicts (i.e. analytical models) 

1551 remove_nodes = [] 

1552 for node, alpha_dict in cost_alloc.items(): 

1553 if len(alpha_dict) == 0: 

1554 remove_nodes.append(node) 

1555 for node in remove_nodes: 

1556 del cost_alloc[node] 

1557 del eval_alloc[node] 

1558 

1559 # Bar chart showing cost allocation breakdown for MF system at final iteration 

1560 fig, ax = plt.subplots(figsize=(6, 5), layout='tight') 

1561 width = 0.7 

1562 x = np.arange(len(cost_alloc)) 

1563 xlabels = list(cost_alloc.keys()) # One bar for each component

1565

1567 bottom = 0 

1568 c_intervals = np.linspace(0, 1, len(alpha_dict))

1577 elif frac > arrow_bar_width: 

1578 xy = (x[j] + width / 2, bottom - frac / 2) # Label smaller bars with a text off to the side 

1579 ax.annotate(f'{alpha}, {round(eval_alloc[node][alpha])}', xy, xytext=(xy[0] + 0.2, xy[1]), 

1580 arrowprops={'arrowstyle': '->', 'linewidth': 1}) 

1581 else: 

1582 pass # Don't label really small bars 

1583 ax_default(ax, '', "Fraction of total cost", legend=False) 

1584 ax.set_xticks(x, xlabels) 

1585 ax.set_xlim(left=-1, right=x[-1] + 1)

1588 fig.savefig(Path(self.root_dir) / 'mf_allocation.pdf', bbox_inches='tight', format='pdf')

1592 def serialize(self, keep_components=False, serialize_args=None, serialize_kwargs=None) -> dict: 

1593 """Convert to a `dict` with only standard Python types for fields. 

1594 

1595 :param keep_components: whether to serialize the components as well (defaults to False) 

1596 :param serialize_args: `dict` of arguments to pass to each component's serialize method 

1597 :param serialize_kwargs: `dict` of keyword arguments to pass to each component's serialize method 

1598 :returns: a `dict` representation of the `System` object 

1599 """ 

1600 serialize_args = serialize_args or dict() 

1601 serialize_kwargs = serialize_kwargs or dict() 

1602 d = {} 

1603 for key, value in self.__dict__.items(): 

1604 if value is not None and not key.startswith('_'): 

1605 if key == 'components' and not keep_components: 

1606 d[key] = [comp.serialize(keep_yaml_objects=False, 

1607 serialize_args=serialize_args.get(comp.name), 

1608 serialize_kwargs=serialize_kwargs.get(comp.name)) 

1609 for comp in value] 

1610 elif key == 'train_history': 

1611 if len(value) > 0: 

1612 d[key] = value.serialize() 

1613 else: 

1614 if not isinstance(value, _builtin): 

1615 self.logger.warning(f"Attribute '{key}' of type '{type(value)}' may not be a builtin " 

1616 f"Python type. This may cause issues when saving/loading from file.") 

1617 d[key] = value 

1618 return d
